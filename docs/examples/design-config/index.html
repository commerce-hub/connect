Example: Design Config

  • Render the widget and assign an InteractionResponse
  • Customize merchant-safe colors, spacing, and copy
  • Bind SDK properties from a Lit host component
  • Create and mount the widget imperatively
  • Bind SDK properties from an Angular 22 component

Design configuration


import 'commerce-hub-connect-sdk';
import type {DesignConfig} from 'commerce-hub-connect-sdk';

const designConfig: DesignConfig = {
  colors: {
    primary: '#084acc',
    text: '#101828',
    copy: '#475467',
    border: '#cbd5e1',
    widgetBackground: '#ffffff',
    paymentMethodBackground: '#f8fafc',
  },
  baseFontSize: '16px',
  baseFontWeight: '400',
  shellPadding: '1.75rem 1.5rem 1.5rem',
  payByBank: {
    apiKey: '<pay-by-bank-api-key>',
    configId: '<pay-by-bank-config-id>',
  },
  copy: {
    otp: {
      title: 'Verify OTP',
      verifyButton: 'Verify',
    },
    paymentMethods: {
      title: 'Choose payment method',
      continueButton: 'Use selected method',
    },
    noSavedPaymentMethods: {
      title: 'Add payment method',
      addCardTitle: 'Add card',
      addBankAccountTitle: 'Add bank account',
    },
  },
};

const widget = document.querySelector('commerce-hub-connect');

if (widget) {
  widget.designConfig = designConfig;
}

Most designConfig fields are optional. Omitted values fall back to SDK defaults. If the Pay by Bank handoff is enabled, provide payByBank.apiKey and payByBank.configId from your CommerceHub integration.
